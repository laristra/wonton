238 template <int D> inline bool
239 approxEq(const Point<D>& p1, const Point<D>& p2, double tol = 1.0e-8) {
240  // This uses the L_infty norm to avoid nearby subtractions
241  for (int i = 0; i < D; i++)
242  if (p1[i] < (p2[i] - tol) || p1[i] > (p2[i] + tol))
243  return false;
244  return true;
245 }

247 template <int D> bool
248 operator<(const Point<D>& p1, const Point<D>& p2) {
249  if (approxEq(p1, p2))
250  return false;
251  else
252  for (int i = 0; i < D; ++i) {
253  if (p1[i] < p2[i])
254  return true;
255  else if (p2[i] < p1[i])
256  return false;
257  }
258  return false;
259 }
