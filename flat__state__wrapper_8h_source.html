430  void mesh_add_data(Entity_kind on_what, std::string const& var_name,
431  std::shared_ptr<std::vector<T>> data) {
432  // if we have seen this entity before - check size match, else
433  // store this size
434  auto sziter = entity_size_map_.find(on_what);
435  if (sziter != entity_size_map_.end()) {
436  if (sziter->second != data->size()) {
437  throw std::runtime_error(std::string("variable ")+var_name+
438  " has incompatible size on add");
439  }
440  } else {
441  entity_size_map_[on_what] = data->size();
442  }
443 
444  // store data and update internal book-keeping
445  pair_t pair(var_name, on_what);
446  auto iter = name_map_.find(pair);
447  if (iter == name_map_.end()) { // have not seen this entity-name combo before, add data
448  state_.push_back(data);
449  name_map_[pair] = state_.size() - 1;
450  entities_.push_back(on_what);
451  entity_map_[var_name] = on_what;
452  entity_size_map_[on_what] = data->size();
453  } else { // have seen the entity-name combo already, replace data. already checked size.
454  std::copy(data->begin(), data->end(), state_[iter->second]->begin());
455  }
456  }
