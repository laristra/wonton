18 /*
19  @file CoordinateSystem.h
20  @brief Defines quantities needed for non-Cartesian coordinates
21 
22  Portage prefers to work with geometry factors of unity (actual volumes rather
23  than wedges). Because of this, a geometry factor is unnecessary. However,
24  to help emphasize to users of Portage that this is the assumption, the
25  geometry factor is listed explicitly for every coordinate system. This also
26  makes it easier to change to different geometry factors in the future if
27  desired.
28 
29  The modify_* routines all take an input calculated using the "standard",
30  Cartesian-like expression, and then modify it to be appropriate for the
31  selected coordinate system. For Cartesian coordinates, this becomes a no-op
32  because the expression is already calculated correctly.
33  */

48  template<int D>
49  static void shift_moments_list_core(
50  std::vector<double> & moments, int const shift,
51  double const scaling_factor) {
52  // Allocate new storage
53  auto top_moment = index_to_moment<D>(moments.size() - 1);
54  auto max_order = std::get<0>(top_moment) - shift;
55  auto num_new_moments = count_moments<D>(max_order);
56  std::vector<double> new_moments(num_new_moments);
57  // Shift moments
58  for (int new_index = 0; new_index < new_moments.size(); new_index++) {
59  auto moment_spec = index_to_moment<D>(new_index);
60  auto order = std::get<0>(moment_spec);
61  auto exponents = std::get<1>(moment_spec);
62  order += shift;
63  exponents[0] += shift;
64  auto old_index = moment_to_index<D>(order, exponents);
65  new_moments[new_index] = moments[old_index];
66  }
67  // Rescale moments
68  for (int d = 0; d < new_moments.size(); ++d) {
69  new_moments[d] *= scaling_factor;
70  }
71  // Swap vectors
72  new_moments.swap(moments);
73  }
74 
75 }

588 struct Spherical3DCoordinates {
589  public:
590 
592  static constexpr double geometry_factor = 1;
593 
595  static constexpr double inv_geo_fac = 1.0 / geometry_factor;
596 
598  template<int D>
599  static constexpr void verify_coordinate_system() {
600  // Valid only in 3D
601  static_assert(D == 3,
602  "Spherical (3D) coordinates only valid in 3D.");
603  }
604 
606  template<int D>
607  static constexpr void modify_gradient(Vector<D> & gradient,
608  Point<D> const & reference_point) {
609  gradient[1] /= reference_point[0];
610  gradient[2] /= (reference_point[0] * sin(reference_point[1]));
611  }
612 
614  template<int D>
615  static constexpr void modify_line_element(Vector<D> & line_element,
616  Point<D> const & reference_point) {
617  line_element[1] *= reference_point[0];
618  line_element[2] *= (reference_point[0] * sin(reference_point[1]));
619  }
620 
623  template<int D>
624  static constexpr void modify_volume(double & volume,
625  Point<D> const & plo, Point<D> const & phi) {
626  // Adjust for different coordinate system
627  auto rbar = 0.5 * (plo[0] + phi[0]);
628  auto dr_2 = 0.5 * (phi[0] - plo[0]);
629  auto thetabar = 0.5 * (phi[1] + plo[1]);
630  auto dtheta_2 = 0.5 * (phi[1] - plo[1]);
631  auto sin_tb = sin(thetabar);
632  auto sinc_dt = sin(dtheta_2) / dtheta_2;
633  volume *= (rbar*rbar + dr_2*dr_2/3.0) * (sin_tb * sinc_dt);
634  // Apply geometry factor
635  volume *= inv_geo_fac;
636  }
637 
640  template<int D>
641  static constexpr void modify_first_moments(Point<D> & moments,
642  Point<D> const & plo, Point<D> const & phi) {
643  // Adjust for different coordinate system
644  auto rbar = 0.5 * (phi[0] + plo[0]);
645  auto dr_2 = 0.5 * (phi[0] - plo[0]);
646  auto rbar_sq = rbar * rbar;
647  auto dr_2_sq = dr_2 * dr_2;
648  auto rr1 = rbar_sq + dr_2_sq;
649  auto rr2 = rbar_sq + dr_2_sq/3.0;
650  auto thetabar = 0.5 * (phi[1] + plo[1]);
651  auto dtheta_2 = 0.5 * (phi[1] - plo[1]);
652  auto sin_tb = sin(thetabar);
653  auto sinc_dt = sin(dtheta_2) / dtheta_2;
654  auto cosc_tb = cos(thetabar) / thetabar;
655  auto cos_dt = cos(dtheta_2);
656  auto ss1 = sin_tb * sinc_dt;
657  moments[0] *= rr1 * ss1;
658  moments[1] *= rr2 * (ss1 + cosc_tb * (sinc_dt - cos_dt));
659  moments[2] *= rr2 * ss1;
660  // Apply geometry factor
661  for (int d = 0; d < D; ++d) {
662  moments[d] *= inv_geo_fac;
663  }
664  }
665 
669  static constexpr int moment_shift = 2;
670 
673  static constexpr double moment_coefficient = 1;
674 
677  template<int D>
678  static void shift_moments_list(std::vector<double> & moments) {
679  // TODO: The precise expression is M^{\rm sph}[i,j,k] = (1/G)
680  // sum_{n=0}^{\infty} \left[ \frac{(-1)^n}{(2n)!} M^{\rm
681  // Cart}[i+2,j+2n,k]\right]. We could allow an approximation that
682  // performs that summation over however many moments are available,
683  // and higher moments will be progressively lower accuracy (because
684  // there will be less moments available for the summation). In that
685  // case, we should still leave moment_shift as 2, because you
686  // definitely lose two orders, and then any order you keep has an
687  // accuracy limited by how many moments are available. But in that
688  // case, we would definitely want the user to be aware of this, such
689  // as by a compiler flag that compiles in the approximate formula vs
690  // compiling in the failed assertion below.
691  // Spherical coordinates include an extra factor of r^2 sin(theta), which
692  // cannot be managed by shifting moments.
693  std::cerr << "The shift_moments_list method does not work in 3D " <<
694  "spherical coordinates." << std::endl;
695  assert(false);
696  }
697 
698 }; // Spherical (3D) Coordinates
