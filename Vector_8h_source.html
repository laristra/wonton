310 Vector<3> cross(const Vector<3>& a, const Vector<3>& b) {
311  Vector<3> r;
312  r[0] = a[1] * b[2] - a[2] * b[1];
313  r[1] = a[2] * b[0] - a[0] * b[2];
314  r[2] = a[0] * b[1] - a[1] * b[0];
315  return r;
