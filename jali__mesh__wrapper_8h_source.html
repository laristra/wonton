41 template <int D>
43 Point<D> toWontonPoint(const JaliGeometry::Point& jp) {
44  Point<D> pp;
45  assert(jp.dim() == D);
46  for (int d = 0; d < D; ++d)
47  pp[d] = jp[d];
48  return pp;
49 }
