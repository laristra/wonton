799  template <class T = double>
800  bool shape_is_good(const std::shared_ptr<StateVectorMulti<T>> sv) {
801 
802  // get the data shape
803  std::unordered_map<int, int> shape{get_material_shape()};
804 
805  // if we don't have cells yet, and don't know the shape, assume the
806  // user knows what they are doing and the shape is good.
807  if (shape.size() == 0) return true;
808 
809  // get the actual data out of the state vector
810  const auto& data = sv->get_data();
811 
812  // this one is tricky, but if we build up the state vector material
813  // we may start with no data, so for the moment, let's pass this
814  if (data.size() == 0) return true;
815 
816  // check first that there are the correct number of materials
817  if (shape.size() != data.size()) return false;
818 
819  for (const auto& kv : data) {
820  if (shape.find(kv.first) == shape.end() ||
821  shape[kv.first] != kv.second.size()
822  ) return false;
823  }
824 
825  return true;
826  }
