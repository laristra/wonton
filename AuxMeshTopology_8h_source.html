1296  double wvol = wedge_volume(wid);
1297

1299  std::array<Point<D>, D+1> wcoords;
1300  wedge_get_coordinates(wid, &wcoords, posvol_order);

1302  wcen += wcoords[i];
1303  wcen /= D+1;

1310  if (vol > std::numeric_limits<double>::epsilon()) *centroid /= vol;
