164 // Check to see if the current cell needs to be refined further
165 template<int D>
166 bool Adaptive_Refinement_Mesh<D>::check_for_refinement(
167  const BoundingBox<D> &cell, const int level) {
168  // Evaluate refinement function
169  // -- In principle one could do something like integrate to get the
170  // average value or find the maximum value, but for now we'll just sample
171  // the center of the cell. This is equivalent to using the average value
172  // to second order for Cartesian geometries (if you use the centroid
173  // instead, you get second-order equivalence on other geometries).
174  Point<D> center;
175  for (int d = 0; d < D; ++d) {
176  center[d] = 0.5 * (cell[d][LO] + cell[d][HI]);
177  }
178  int value = refinement_level_(center);
179  // Some AMR grids enforce further restrictions, such as the requirement that
180  // adjacent cells differ by no more than a single level of refinement, or
181  // that you need to refine in larger patches rather than by cell, etc. This
182  // is a simple mesh, and by not imposing these constraints it actually makes
183  // this mesh more general.
184  return(value > level);
185 }

187 // ____________________________________________________________________________
188 // Recursive procedure to perform the actual splitting of a cell by axis.
189 // -- Refinement is done by splitting in half along each axis.
190 template<int D>
191 typename Adaptive_Refinement_Mesh<D>::mesh_data_t
192  Adaptive_Refinement_Mesh<D>::split_cell(
193  const int d, const BoundingBox<D> &cell) {
194  // Splitting point
195  double midpoint = 0.5 * (cell[d][LO] + cell[d][HI]);
196  // Create the lower cell
197  BoundingBox<D> cell_lo;
198  for (int d2 = 0; d2 < D; ++d2) {
199  cell_lo[d2][LO] = cell[d2][LO];
200  cell_lo[d2][HI] = cell[d2][HI];
201  }
202  cell_lo[d][HI] = midpoint;
203  // Create the upper cell
204  BoundingBox<D> cell_hi;
205  for (int d2 = 0; d2 < D; ++d2) {
206  cell_hi[d2][LO] = cell[d2][LO];
207  cell_hi[d2][HI] = cell[d2][HI];
208  }
209  cell_hi[d][LO] = midpoint;
210  // Compose all the data together
211  mesh_data_t newcells;
212  const int N = 1<<(d+1);
213  newcells.resize(N);
214  if (d > 0) { // Recursive case
215  // Split lower cell along other axes
216  auto tempcells = split_cell(d-1, cell_lo);
217  for (int n = 0; n < N/2; ++n) {
218  newcells[n] = tempcells[n];
219  }
220  // Split upper cell along other axes
221  tempcells = split_cell(d-1, cell_hi);
222  for (int n = 0; n < N/2; ++n) {
223  newcells[n+N/2] = tempcells[n];
224  }
225  } else { // Base case
226  // Return the two new cells
227  newcells[0] = cell_lo;
228  newcells[1] = cell_hi;
229  }
230  return(std::move(newcells));
231 }

233 // ____________________________________________________________________________
234 // Refine a single cell.
235 // -- Remove the specified cell and replace it with refined cells.
236 // -- Some AMR meshes preserve non-leaf cells, but this mesh throws them
237 // away as they are irrelevant to mapping to another grid.
238 template<int D>
239 std::pair<typename Adaptive_Refinement_Mesh<D>::mesh_data_t, std::vector<int>>
240  Adaptive_Refinement_Mesh<D>::refine_cell(
241  const Adaptive_Refinement_Mesh<D>::mesh_data_t &cells,
242  const std::vector<int> &level, const int n) {
243  assert(n >= 0);
244  assert(n < cells.size());
245  // Create new storage
246  // -- New storage replaces 1 cell with 2^D cells
247  const int num_new_cells = (1<<D) - 1;
248  const int newsize = cells.size() + num_new_cells;
249  mesh_data_t newcells;
250  std::vector<int> newlevel;
251  newcells.resize(newsize);
252  newlevel.resize(newsize);
253  // Copy elements prior to the one to refine
254  for (int i = 0; i < n; ++i) {
255  newcells[i] = cells[i];
256  newlevel[i] = level[i];
257  }
258  // Replace specified cell with new cells
259  mesh_data_t tempcells = split_cell(D-1,cells[n]);
260  for (int i = 0; i < tempcells.size(); ++i) {
261  newcells[n+i] = tempcells[i];
262  newlevel[n+i] = level[n] + 1;
263  }
264  // Copy elements after the one to refine
265  for (int i = n+1; i < level.size(); ++i) {
266  newcells[i+num_new_cells] = cells[i];
267  newlevel[i+num_new_cells] = level[i];
268  }
269  // Return
270  return(std::move(std::make_pair(newcells, newlevel)));
271 }

273 // ____________________________________________________________________________
274 // Build the mesh based on the mesh corners and the refinement function.
275 template<int D>
276 void Adaptive_Refinement_Mesh<D>::build_mesh() {
277  // Build the level-zero grid
278  cells_.resize(1);
279  cells_[0] = mesh_corners_;
280  std::vector<int> level = {0};
281  // Refinement loop
282  bool do_refinement;
283  for (int n = 0; n < level.size(); ++n) {
284  if (check_for_refinement(cells_[n], level[n])) {
285  std::tie(cells_, level) = refine_cell(cells_, level, n);
286  --n; // Decrement to repeat the current cell, as it was replaced
287  }
288  }
289 }
