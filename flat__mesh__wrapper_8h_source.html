169  // Create complete topologies for the various mesh entities.This topology is
170  // defined by cells, faces and nodes. This is true in both dimensions 2 and 3
171  // but obviously what is meant by these changes with dimension.
172  // These topologies include the maps: cellToFace and faceToNode
173  // and the skip map: cellToNode.
174  // We also create the single inverse map going from nodeToCell.
175  // This routine also computes offsets, which turn counts into partial sums so
176  // that the relevant topologies, e.g. cell face indices, can be extracted from
177  // the corresponding complete lists
178  // Since faces are unique between cells, we also need to assign a direction to
179  // each face in a cell
180  void make_index_maps() {
181 
182  if (dim_ == 2)
183  {
184  // In 2D, we need to construct faces and their connectivity since the
185  // boundary representation is just vertices around cells. A face in 2D is
186  // and edge, and edges aren't specified, so we need to create them. Faces
187  // also need to have directions and we need to make sure those are created
188  // as well.
189  // In 2D we are given the skip topology cellToNode, and we need to construct
190  // the adjacent topologies cellToFace and faceToNode.
191 
192  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
193 
194  // Clear and reserve
195  cellToFaceList_.clear();
196  cellToFaceList_.reserve(cellNodeCounts_.size());
197 
198  cellToFaceDirs_.clear();
199  cellToFaceDirs_.reserve(cellNodeCounts_.size());
200 
201  faceToNodeList_.clear();
202  faceToNodeList_.reserve(cellToNodeList_.size()); // slight underestimate
203 
204  // in 2D we directly specify the cell to node map. In order to complete
205  // the topology we need to construct faces on the fly. Each face is
206  // defined by an (vertex, next_vertex) pair. There are as many faces(edges)
207  // as vertices.
208 
209  // a temporary map that takes an edge node pair (v, next_v) and maps it to
210  // a face id
211  std::map<std::pair<int, int>, int> nodePairToFaceTmp;
212 
213  // new face/edge counter
214  int facecount = 0;
215 
216  for (unsigned int c=0; c<cellNodeCounts_.size(); ++c) {
217  int offset = cellNodeOffsets_[c];
218  int count = cellNodeCounts_[c];
219  for (unsigned int i=0; i<count; ++i) {
220  int n0 = cellToNodeList_[offset+i];
221  int n1 = cellToNodeList_[offset+((i+1)%count)];
222  // put nodes in canonical order
223  int p0 = std::min(n0, n1);
224  int p1 = std::max(n0, n1);
225  auto npair = std::make_pair(p0, p1);
226  auto it = nodePairToFaceTmp.find(npair);
227  int face;
228  if (it == nodePairToFaceTmp.end()) {
229  // new face
230  face = facecount;
231  nodePairToFaceTmp.insert(std::make_pair(npair, face));
232  faceToNodeList_.emplace_back(p0);
233  faceToNodeList_.emplace_back(p1);
234  ++facecount;
235  }
236  else {
237  // existing face
238  face = it->second;
239  }
240  cellToFaceList_.emplace_back(face);
241  cellToFaceDirs_.push_back(n0 == p0);
242  } // for i
243 
244  if (c == numOwnedCells_ - 1) numOwnedFaces_ = facecount;
245 
246  } // for c
247  }
248  else if (dim_ == 3)
249  {
250  // In 3D, we are given the adjacent topologies cellToFace and faceToNode
251  // and we only need to compute the skip topology cellToNode
252 
253  compute_offsets(faceNodeCounts_, &faceNodeOffsets_);
254  compute_offsets(cellFaceCounts_, &cellFaceOffsets_);
255 
256  // Compute cell-to-node adjacency lists (3D only)
257  cellNodeCounts_.clear();
258  cellNodeCounts_.reserve(cellFaceCounts_.size());
259 
260  cellToNodeList_.clear();
261  cellToNodeList_.reserve(cellFaceCounts_.size() * 4);
262 
263  for (unsigned int c=0; c<cellFaceCounts_.size(); ++c) {
264  std::vector<int> cellfaces, dummy;
265  cell_get_faces_and_dirs(c, &cellfaces, &dummy);
266  std::set<int> cellnodes;
267 
268  for (auto const& f : cellfaces) {
269  std::vector<int> facenodes;
270  face_get_nodes(f, &facenodes);
271  cellnodes.insert(facenodes.begin(), facenodes.end());
272  }
273 
274  cellNodeCounts_.emplace_back(cellnodes.size());
275  cellToNodeList_.insert(cellToNodeList_.end(),
276  cellnodes.begin(), cellnodes.end());
277  }
278  compute_offsets(cellNodeCounts_, &cellNodeOffsets_);
279  }
280 
281  // Compute inverse node-to-cell adjacency lists
282  int numNodes = nodeCoords_.size()/dim_;
283  std::vector<std::set<int>> nodeToCellTmp(numNodes);
284  for (unsigned int c=0; c<cellNodeCounts_.size(); ++c) {
285  int offset = cellNodeOffsets_[c];
286  int count = cellNodeCounts_[c];
287  for (unsigned int i=0; i<count; ++i) {
288  int n = cellToNodeList_[offset+i];
289  nodeToCellTmp[n].insert(c);
290  }
291  }
292 
293  nodeCellCounts_.clear();
294  nodeCellCounts_.reserve(numNodes);
295 
296  nodeToCellList_.clear();
297  nodeToCellList_.reserve(cellToNodeList_.size());
298 
299  for (unsigned int n=0; n<numNodes; ++n) {
300  const std::set<int>& nodes = nodeToCellTmp[n];
301  nodeCellCounts_.emplace_back(nodes.size());
302  nodeToCellList_.insert(nodeToCellList_.end(), nodes.begin(), nodes.end());
303  }
304 
305  compute_offsets(nodeCellCounts_, &nodeCellOffsets_);
306 
307  } // make_index_maps

310  int compute_offsets(const std::vector<int>& counts,
311  std::vector<int>* offsets) {
312  offsets->resize(counts.size());
313  if (counts.size()) {
314  (*offsets)[0] = 0;
315  std::partial_sum(counts.begin(),
316  counts.end()-1,
317  offsets->begin()+1);
318  }
