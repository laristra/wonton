238 // ____________________________________________________________________________
239 // Get lower and upper corners of cell bounding box
240 template<int D, class CoordSys>
241 void Direct_Product_Mesh_Wrapper<D,CoordSys>::cell_get_bounds(
242  const int id, Point<D> *plo, Point<D> *phi) const {
243  std::array<int,D> indices = cellid_to_indices(id);
244  // Cell axis points are zero-indexed, cells are zero-indexed. Thus cell 0 is
245  // bounded by axis points 0 and 1, or more generally, cell N is bounded by
246  // axis points N and N+1.
247  for (int d = 0; d < D; ++d) {
248  (*plo)[d] = mesh_.get_axis_point(d, indices[d]);
249  (*phi)[d] = mesh_.get_axis_point(d, indices[d]+1);
250  }
251 }

284 // ____________________________________________________________________________
285 // Convert from indices to ID
286 template<int D, class CoordSys>
287 int Direct_Product_Mesh_Wrapper<D,CoordSys>::indices_to_cellid(
288  const std::array<int,D>& indices) const {
289  assert(D == mesh_.space_dimension());
290  int id = 0;
291  // Loop over all but the last dimension
292  for (int d = D-1; d > 0; --d) {
293  int idx = indices[d];
294  int mult = axis_num_cells(d-1);
295  id += idx;
296  id *= mult;
297  }
298  // Handle the last dimension
299  id += indices[0];
300  // Return
301  return id;
302 }
303 
304 // ____________________________________________________________________________
305 // Convert from ID to indices
306 template<int D, class CoordSys>
307 std::array<int,D> Direct_Product_Mesh_Wrapper<D,CoordSys>::cellid_to_indices(
308  const int id) const {
309  assert(D == mesh_.space_dimension());
310  std::array<int,D> indices;
311  int residual = id;
312  // Construct the denominators
313  std::array<int,D> denom;
314  denom[0] = 1;
315  for (int d = 1; d < D; ++d) {
316  denom[d] = denom[d-1] * axis_num_cells(d-1);
317  }
318  // Loop over all but the last dimension
319  for (int d = D-1; d > 0; --d) {
320  int index = residual / denom[d];
321  residual -= index * denom[d];
322  indices[d] = (int) index;
323  }
324  // Handle the last dimension
325  indices[0] = (int) residual;
326  // Return
327  return indices;
328 }
